    # Boyer-Moore Voting Algorithm

    leetcode 169. Majority Element

    /**
 * @param {number[]} nums
 * @return {number}
 *
 *                    *
 *          *       *
 *    *   *   *   *
 *  *   *       *
 *
 */
var majorityElement = function(nums) {
    let majority = null;
    let count = 0;
    for (let i = 0; i < nums.length; i++) {
        if (count === 0) {
            majority = nums[i];
            count = 1;
        } else {
            if (majority === nums[i]) {
                count += 1;
            } else {
                count -= 1;
            }
        }
    }

    return majority;
    
};
